    <Wrapper>

        <button className="btn reset-btn" onClick={() => setCount(0)}>
          reset
        </button>
        <button className="btn inc-btn" onClick={() => setCount(count + 1)}>
          increase
        </button>
      </div>
    </Wrapper>
  )
